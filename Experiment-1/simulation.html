Query Examples

Basic COUNT

SELECT COUNT(*) 
FROM employees;

COUNT with GROUP BY

SELECT department, 
       COUNT(*) AS count
FROM employees 
GROUP BY department;

AVG with GROUP BY

SELECT department, 
       AVG(salary) AS avg_salary
FROM employees 
GROUP BY department;

HAVING Clause

SELECT department, 
       COUNT(*) AS count
FROM employees 
GROUP BY department 
HAVING COUNT(*) > 2;
